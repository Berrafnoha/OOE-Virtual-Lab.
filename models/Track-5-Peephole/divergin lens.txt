// ─── DRAW HELPERS ───
function resize() {
  const area = canvas.parentElement;
  canvas.width = area.clientWidth;
  canvas.height = area.clientHeight || 300;
}

function drawLabel(x, y, text, color, align='center', size=11) {
  ctx.save();
  ctx.font = `500 ${size}px 'DM Sans', sans-serif`;
  ctx.fillStyle = color;
  ctx.textAlign = align;
  ctx.fillText(text, x, y);
  ctx.restore();
}

function drawArrow(x1, y1, x2, y2, color, width=2, dashed=false) {
  ctx.save();
  ctx.strokeStyle = color;
  ctx.lineWidth = width;
  ctx.lineCap = 'round';
  if (dashed) ctx.setLineDash([6, 4]);
  ctx.beginPath();
  ctx.moveTo(x1, y1);
  ctx.lineTo(x2, y2);
  ctx.stroke();
  // arrowhead
  if (!dashed) {
    const angle = Math.atan2(y2-y1, x2-x1);
    ctx.setLineDash([]);
    ctx.fillStyle = color;
    ctx.beginPath();
    ctx.moveTo(x2, y2);
    ctx.lineTo(x2 - 10*Math.cos(angle-0.4), y2 - 10*Math.sin(angle-0.4));
    ctx.lineTo(x2 - 10*Math.cos(angle+0.4), y2 - 10*Math.sin(angle+0.4));
    ctx.closePath();
    ctx.fill();
  }
  ctx.restore();
}

function drawLens(cx, top, bottom, color) {
  const W = canvas.width, H = canvas.height;
  // draw diverging lens shape (two arcs curving inward)
  ctx.save();
  ctx.strokeStyle = color;
  ctx.lineWidth = 2.5;
  ctx.fillStyle = 'rgba(0,212,232,0.06)';

  const h = bottom - top;
  const bulge = Math.min(18, h * 0.18);

  ctx.beginPath();
  ctx.moveTo(cx, top);
  ctx.quadraticCurveTo(cx + bulge, top + h/2, cx, bottom);
  ctx.quadraticCurveTo(cx - bulge, top + h/2, cx, top);
  ctx.fill();
  ctx.stroke();

  // arrows top/bottom
  ctx.fillStyle = color;
  // top arrow
  ctx.beginPath(); ctx.moveTo(cx, top-2); ctx.lineTo(cx-5, top+10); ctx.lineTo(cx+5, top+10); ctx.closePath(); ctx.fill();
  // bottom arrow
  ctx.beginPath(); ctx.moveTo(cx, bottom+2); ctx.lineTo(cx-5, bottom-10); ctx.lineTo(cx+5, bottom-10); ctx.closePath(); ctx.fill();

  ctx.restore();
}

function drawAxis(y, x1, x2) {
  ctx.save();
  ctx.strokeStyle = C.axis;
  ctx.lineWidth = 1;
  ctx.setLineDash([6, 5]);
  ctx.beginPath(); ctx.moveTo(x1, y); ctx.lineTo(x2, y); ctx.stroke();
  ctx.restore();
}

function drawFocalPoint(x, y, label, color) {
  ctx.save();
  ctx.fillStyle = color;
  ctx.beginPath(); ctx.arc(x, y, 5, 0, Math.PI*2); ctx.fill();
  drawLabel(x, y+18, label, color, 'center', 11);
  ctx.restore();
}

// ─── PEEPHOLE DRAWINGS ───
function drawPeephole0() {
  const W = canvas.width, H = canvas.height;
  const mid = H/2, lx = W*0.45;

  ctx.clearRect(0,0,W,H);
  drawAxis(mid, 20, W-20);

  // door
  ctx.save();
  ctx.fillStyle = 'rgba(100,120,140,0.12)';
  ctx.strokeStyle = 'rgba(100,120,140,0.3)';
  ctx.lineWidth = 2;
  ctx.fillRect(lx - 35, 20, 70, H-40);
  ctx.strokeRect(lx - 35, 20, 70, H-40);
  ctx.restore();
  drawLabel(lx, 16, 'DOOR', C.dim, 'center', 10);

  // wide rays coming in (no lens yet)
  const rays = [
    { y1: H*0.1, angle: 35 },
    { y1: H*0.3, angle: 15 },
    { y1: mid,   angle: 0  },
    { y1: H*0.7, angle: -15},
    { y1: H*0.9, angle: -35},
  ];
  rays.forEach(r => {
    const x1 = 20;
    const x2 = lx - 35;

function drawLaser2() {
  const W=canvas.width, H=canvas.height, mid=H/2;
  const lx1=W*0.3, lx2=W*0.6;
  ctx.clearRect(0,0,W,H);
  drawAxis(mid, 10, W-10);

  ctx.save();
  ctx.fillStyle='rgba(0,230,118,0.15)'; ctx.strokeStyle=C.green; ctx.lineWidth=1.5;
  roundRect(ctx, 10, mid-14, 60, 28, 6);
  ctx.fill(); ctx.stroke();
  drawLabel(40, mid+5, 'LASER', C.green, 'center', 11);
  ctx.restore();

  // narrow in
  ctx.save(); ctx.strokeStyle=C.green; ctx.lineWidth=4; ctx.lineCap='round';
  ctx.beginPath(); ctx.moveTo(70, mid-3); ctx.lineTo(lx1, mid-3); ctx.stroke();
  ctx.beginPath(); ctx.moveTo(70, mid+3); ctx.lineTo(lx1, mid+3); ctx.stroke();
  ctx.restore();

  drawLens(lx1, mid-60, mid+60, C.lens);
  drawLabel(lx1, mid+76, 'Diverging', C.lens, 'center', 10);

  // diverge
  ctx.save(); ctx.strokeStyle=C.green; ctx.lineWidth=2; ctx.lineCap='round';
  ctx.beginPath(); ctx.moveTo(lx1, mid-3); ctx.lineTo(lx2, mid-40); ctx.stroke();
  ctx.beginPath(); ctx.moveTo(lx1, mid); ctx.lineTo(lx2, mid); ctx.stroke();
  ctx.beginPath(); ctx.moveTo(lx1, mid+3); ctx.lineTo(lx2, mid+40); ctx.stroke();
  ctx.restore();

  // converging lens (bigger)
  ctx.save();
  ctx.strokeStyle=C.amber; ctx.lineWidth=2.5; ctx.fillStyle='rgba(245,166,35,0.07)';
  const ct=mid-75, cb=mid+75;
  ctx.beginPath(); ctx.moveTo(lx2,ct); ctx.quadraticCurveTo(lx2+20,mid,lx2,cb); ctx.quadraticCurveTo(lx2-20,mid,lx2,ct); ctx.fill(); ctx.stroke();
  ctx.fillStyle=C.amber;
  ctx.beginPath(); ctx.moveTo(lx2-5,ct+10); ctx.lineTo(lx2,ct-2); ctx.lineTo(lx2+5,ct+10); ctx.fill();
  ctx.beginPath(); ctx.moveTo(lx2-5,cb-10); ctx.lineTo(lx2,cb+2); ctx.lineTo(lx2+5,cb-10); ctx.fill();
  ctx.restore();
  drawLabel(lx2, cb+16, 'Converging', C.amber, 'center', 10);

  drawArrow(lx2, mid-40, W-20, mid-40, C.green, 3);
  drawArrow(lx2, mid, W-20, mid, C.green, 3);
  drawArrow(lx2, mid+40, W-20, mid+40, C.green, 3);

  drawLabel(W*0.82, mid-54, 'Wide parallel', C.green, 'center', 10);
  drawLabel(W*0.82, mid-43, 'beam out', C.green, 'center', 10);
}

// ─── EYE HELPER ───
function drawEye(cx, cy) {
  ctx.save();
  ctx.strokeStyle='rgba(255,255,255,0.3)'; ctx.lineWidth=1.5;
  ctx.fillStyle='rgba(255,255,255,0.05)';
  ctx.beginPath(); ctx.ellipse(cx, cy, 10, 14, 0, 0, Math.PI*2); ctx.fill(); ctx.stroke();
  ctx.fillStyle='#1B98A0';
  ctx.beginPath(); ctx.arc(cx, cy, 5, 0, Math.PI*2); ctx.fill();
  ctx.restore();
}

function roundRect(ctx, x, y, w, h, r) {
  ctx.beginPath();
  ctx.moveTo(x+r, y);
  ctx.lineTo(x+w-r, y); ctx.quadraticCurveTo(x+w,y,x+w,y+r);
  ctx.lineTo(x+w, y+h-r); ctx.quadraticCurveTo(x+w,y+h,x+w-r,y+h);
  ctx.lineTo(x+r, y+h); ctx.quadraticCurveTo(x,y+h,x,y+h-r);
  ctx.lineTo(x, y+r); ctx.quadraticCurveTo(x,y,x+r,y);
  ctx.closePath();
}

// ─── RENDER STEP ───
function renderStep() {
  resize();
  const app = APPS[currentApp];
  const step = app.steps[currentStep];
  step.draw && step.draw();
  updateUI();
}

function updateUI() {
  const app = APPS[currentApp];
  const step = app.steps[currentStep];
  const total = app.steps.length;

  document.getElementById('canvas-title').textContent = app.title;

  const badge = document.getElementById('step-badge');
  badge.textContent = `STEP ${currentStep+1} / ${total}`;
  badge.style.background = app.badge.bg;
  badge.style.color = app.badge.color;

  // dots
  const dotsEl = document.getElementById('step-dots');
  dotsEl.innerHTML = '';
  for(let i=0; i<total; i++){
    const d = document.createElement('div');
    d.className = 'step-dot' + (i < currentStep ? ' done' : i===currentStep ? ' active' : '');
    d.onclick = () => { currentStep=i; renderStep(); };
    dotsEl.appendChild(d);
  }

  document.getElementById('prev-btn').disabled = currentStep===0;
  document.getElementById('next-btn').disabled = currentStep===total-1;
  document.getElementById('next-btn').textContent = currentStep===total-1 ? 'Done ✓' : 'Next →';

  // info card
  const card = document.getElementById('info-card');
  const factClass = step.factType ? ` ${step.factType}` : '';
  card.innerHTML = `
    <div class="info-step-num" style="color:${app.color}">STEP ${currentStep+1} — ${currentApp.toUpperCase()}</div>
    <div class="info-title">${step.title}</div>
    <div class="info-body">${step.body}</div>
    ${step.fact ? `<div class="key-fact${factClass}">${step.fact}</div>` : ''}
  `;
}

function nextStep() {
  const total = APPS[currentApp].steps.length;
  if(currentStep < total-1){ currentStep++; renderStep(); }
}
function prevStep() {
  if(currentStep > 0){ currentStep--; renderStep(); }
}

function switchApp(app) {
  currentApp = app;
  currentStep = 0;

  document.querySelectorAll('.app-tab').forEach(t => t.classList.toggle('active', t.dataset.app===app));
  document.querySelectorAll('.overview-item').forEach(t => t.classList.toggle('active', t.dataset.app===app));

  renderStep();
}

// Init
window.addEventListener('resize', () => renderStep());
renderStep();
